## ðŸŽ¨ Code Examples for Reference

### Creating a New Dashboard Widget
```tsx
// Template: src/components/dashboard/ExampleWidget.tsx
import React, { useEffect, useState } from 'react';
import { useRouter } from 'expo-router';
import { useAuth } from '../../contexts/AuthContext';
import { dbHelpers } from '../../services/supabase';

export const ExampleWidget: React.FC = () => {
  const router = useRouter();
  const { user } = useAuth();
  const [data, setData] = useState(null);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    if (user) {
      loadData();
    }
  }, [user]);

  const loadData = async () => {
    try {
      setIsLoading(true);
      const { data, error } = await dbHelpers.table.getAll();
      if (error) throw error;
      setData(data);
    } catch (error) {
      console.error('Error:', error);
    } finally {
      setIsLoading(false);
    }
  };

  if (isLoading) {
    return <div>Loading...</div>;
  }

  return (
    <div className="bg-white/5 backdrop-blur-xl rounded-2xl border border-white/10 p-6">
      {/* Widget content */}
    </div>
  );
};
```

### Adding a Database Query
```typescript
// In src/services/supabase.ts
export const dbHelpers = {
  // ... existing helpers
  newTable: {
    getAll: async () => {
      return await supabase
        .from('table_name')
        .select('*')
        .order('created_at', { ascending: false });
    },
    getById: async (id: string) => {
      return await supabase
        .from('table_name')
        .select('*')
        .eq('id', id)
        .single();
    },
  },
};
```

## ðŸš¨ Common Pitfalls to Avoid

1. **Don't modify database schema without migration**
   - Always create a new migration file
   - Test locally before deploying

2. **Don't forget RLS policies**
   - Every new table needs RLS enabled
   - Add policies for user data isolation

3. **Don't skip TypeScript types**
   - Define interfaces for all data structures
   - Use strict type checking

4. **Don't bypass the service layer**
   - Always use dbHelpers, never raw Supabase client in components
   - Centralize API calls in service modules

5. **Don't forget loading/error states**
   - Every async operation needs loading state
   - Always handle errors gracefully
